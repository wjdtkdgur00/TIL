# 버블정렬
버블정렬이 뭔지 잘 모르겠으니 일단 예시 코드만 작성하고 추후에 더 자세히 알아오겠다.
```
#include <stdio.h>
#pragma warning(disable:4996)
int main(){
	//버블정렬
	int arr[5] = { 29,10,14,37,13 };
	int i, j, temp;
	for (i = 0; i < 4; i++) {
		for (j = 0; j < 4; j++) {
			if (arr[j] > arr[j + 1]) {
				temp = arr[j];
				arr[j] = arr[j + 1];
				arr[j + 1] = temp;
			}
		}
	}
	for (i = 0; i < 5; i++) {
		printf("%d ", arr[i]);
	}

	//선택정렬
	//모르게써요..ㅠㅠ


	return 0;
}
```
